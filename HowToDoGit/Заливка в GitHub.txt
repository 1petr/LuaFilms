git init                            			 	 //Создание репозитория

git status                          			 	 //Проверить текущее состояние репозитория

git add .                             				 //Индексация изменений файлов

git reset							 //Снять индексацию
						
git commit -m "First Commit"          				 //Сделать коммит с названием First Commit

git remote                                                       //Имена удаленных репозиториев

git branch -a							 //Получить список веток

git remote add origin https://github.com/Vaper-/ONE              //Подключение к удаленному репозиторию

git push -u origin master                                        //Отправка изменений  

git pull                                                         //извлекает (fetch) данные с сервера, с которого вы изначально склонировали, и автоматически пытается слить (merge) их с кодом, над которым вы в данный момент работаете


//Нужно создавать всегда новую ветку
git checkout -b <Имя_Ветки>

//Затем отправлять изменения на github
$ git push origin <Имя_Ветки>

//Если после команды git status написано, что некоторые 
файлы были удалены (deleted <file>), то нужно обязательно
проиндексировать все удаленные файлы командой git add <file>
